    function showCustomAlert(msg, duration = 4000) {

    // --- FUNÇÃO DE DESENHO (CORTANDO IMAGENS EM CÍRCULOS) ---
    const drawChar = (type, x, y, isDucking, currentFrames) => {
        ctx.save();
        ctx.translate(x, y); 
        
        // Efeito visual de dano
        if (isInvincible && !isDancing && !isFlying && currentFrames % 10 < 5) ctx.globalAlpha = 0.5;

        // Voo ou Abaixar
        if (isFlying || type === 'lucas') {
            // Asinhas enquanto voa (ou para o Lucas, que sempre voa)
            ctx.fillStyle = 'rgba(255,255,255,0.8)';
            ctx.beginPath(); ctx.ellipse(-20, -30, 15, 8, Math.PI/4, 0, Math.PI*2); ctx.fill();
            ctx.beginPath(); ctx.ellipse(20, -30, 15, 8, -Math.PI/4, 0, Math.PI*2); ctx.fill();
        } else if (isDucking) {
            ctx.scale(1, 0.6); 
        }

        // Sombra se não estiver voando
        if (!isFlying && type !== 'lucas') {
            ctx.fillStyle = 'rgba(0,0,0,0.2)';
            ctx.beginPath(); ctx.ellipse(0, 0, 15, 4, 0, 0, Math.PI * 2); ctx.fill();
        }

        if (assets[type] && assets[type].complete && assets[type].naturalWidth !== 0) {
            // Recorte Circular Mágico para remover o quadriculado
            ctx.beginPath();
            ctx.arc(0, -30, 25, 0, Math.PI * 2);
            ctx.closePath();
            ctx.clip(); // Corta tudo fora do círculo
            
            // Desenha a imagem centralizada no círculo
            ctx.drawImage(assets[type], -25, -55, 50, 50); 
            ctx.restore();
            return; 
        }

        // Fallback (Desenhos antigos caso a imagem não carregue)
        ctx.fillStyle = type === 'mahalia' ? '#81d4fa' : type === 'mango' ? '#2ecc71' : type === 'lucas' ? '#f1c40f' : '#222';
        ctx.beginPath(); ctx.arc(0, -30, 20, 0, Math.PI*2); ctx.fill();
        ctx.restore();
    };

    class Cloud {
        constructor() {
            this.x = WIDTH + Math.random() * 300; this.y = 20 + Math.random() * 80;
            this.speed = gameSpeed * 0.1 + Math.random() * 0.2; this.size = 20 + Math.random() * 15;
        }
        update() { this.x -= this.speed; }
        draw() {
            ctx.fillStyle = 'rgba(255, 255, 255, 0.85)';
            ctx.beginPath(); ctx.arc(this.x, this.y, this.size, 0, Math.PI*2); ctx.fill();
            ctx.beginPath(); ctx.arc(this.x + this.size*0.8, this.y - this.size*0.4, this.size*0.8, 0, Math.PI*2); ctx.fill();
            ctx.beginPath(); ctx.arc(this.x + this.size*1.6, this.y, this.size*0.9, 0, Math.PI*2); ctx.fill();
        }
    }

    class Player {
        constructor() {
            this.x = 220; this.groundY = 320; 
            this.y = this.groundY; 
            this.width = 30; this.height = 50; 
            this.dy = 0; this.jumpPower = -14; this.gravity = 0.7; 
            this.isGrounded = true; this.isDucking = false;
            this.history = []; 
        }

        jump() { if (this.isGrounded && !this.isDucking && !isFlying) { this.dy = this.jumpPower; this.isGrounded = false; } }
        duck(status) { if(!isFlying) { this.isDucking = status; this.height = status ? 25 : 50; } }

        update() {
            if (isFlying) {
                // Flutua no ar suavemente
                let targetY = this.groundY - 120 + Math.sin(frames * 0.1) * 10;
                this.y += (targetY - this.y) * 0.1;
                this.dy = 0; this.isGrounded = false; this.isDucking = false;
            } else {
                this.y += this.dy;
                if (this.y < this.groundY) { this.dy += this.gravity; this.isGrounded = false; } 
                else { this.y = this.groundY; this.dy = 0; this.isGrounded = true; }
            }

            this.history.push({ x: this.x, y: this.y, ducking: this.isDucking, flying: isFlying });
            if (this.history.length > 80) this.history.shift(); 
        }

        draw() {
            if (isDancing) {
                ctx.save(); ctx.fillStyle = `hsla(${frames * 10 % 360}, 100%, 60%, 0.5)`;
                ctx.beginPath(); ctx.ellipse(this.x, this.y - 25, 40, 50, 0, 0, Math.PI*2); ctx.fill(); ctx.restore();
            }
            drawChar('mahalia', this.x, this.y, this.isDucking, frames);
        }
    }

    class Companion {
        constructor(type, delayIndex) { this.type = type; this.delay = delayIndex; }
        draw(playerHistory) {
            if (this.type === 'lucas' && player) {
                // Lucas voa flutuando em cima da Mahalia
                let targetX = player.x + 30; // Um pouco atrás e pro lado
                let targetY = player.y - 70 + Math.sin(frames * 0.1) * 10;
                drawChar(this.type, targetX, targetY, false, frames);
            } else {
                const index = Math.max(0, playerHistory.length - 1 - this.delay);
                if (index >= 0 && playerHistory[index]) {
                    const pos = playerHistory[index];
                    drawChar(this.type, pos.x - (this.delay * 4), pos.y, pos.ducking, frames);
                }
            }
        }
    }

    class Obstacle {
        constructor() {
            const rand = Math.random();
            if (rand < branchChance && distance > 50) {
                this.type = currentMap === 'surubim' ? 'bird' : 'branch';
                this.y = 285; this.height = 30; this.width = 60; this.speed = gameSpeed + 1;
            } else if (rand < 0.7) {
                this.type = 'car';
                this.y = 320; this.height = 40; this.width = 70;
                this.emoji = Math.random() > 0.5 ? '🚗' : '🚙'; this.speed = gameSpeed + 1;
            } else {
                this.type = 'moto';
                this.y = 320; this.height = 35; this.width = 50;
                this.emoji = '🏍️'; this.speed = gameSpeed + 2.5; 
            }
            this.x = WIDTH + 100;
        }
        update() { this.x -= this.speed; }
        draw() {
            if (this.type === 'branch') {
                ctx.fillStyle = '#4e342e'; ctx.fillRect(this.x + 20, 150, 20, 200); 
                ctx.fillStyle = '#5d4037'; ctx.fillRect(this.x - 20, this.y - this.height, this.width + 30, this.height); 
                ctx.fillStyle = '#2ecc71';
                ctx.beginPath(); ctx.arc(this.x + 20, this.y - 20, 30, 0, Math.PI*2); ctx.fill();
                ctx.beginPath(); ctx.arc(this.x - 10, this.y - 25, 25, 0, Math.PI*2); ctx.fill();
            } else if (this.type === 'bird') {
                ctx.font = '45px Arial'; ctx.textAlign = 'center'; ctx.fillText('🦅', this.x, this.y);
            } else {
                ctx.font = (this.type === 'car' ? '55px' : '45px') + ' Arial';
                ctx.textAlign = 'center'; ctx.textBaseline = 'bottom'; ctx.fillText(this.emoji, this.x, this.y);
            }
        }
    }

    class BackgroundTree {

        }
    }

    function init() {

    function spawn() {
        if (frames % 40 === 0 && currentMap === 'avenida') bgTrees.push(new BackgroundTree());
        if (frames % 100 === 0) clouds.push(new Cloud());
        let spawnRate = Math.max(35, 90 - Math.floor(distance / 15));
        if (frames % spawnRate === 0) obstacles.push(new Obstacle());
    }

    function handleProgression() {
        let d = Math.floor(distance);
        
        if (d >= 100 && !companions.find(c => c.type === 'benny')) {
            companions.push(new Companion('benny', 12)); lives++; showCustomAlert("🐶 Seu fiel escudeiro Benny chegou!");
        }
        if (d >= 300 && !companions.find(c => c.type === 'mango')) {
            companions.push(new Companion('mango', 24)); lives++;
            colDance.style.display = 'flex'; btnDance.style.display = 'flex';
            showCustomAlert("🕺 Seu melhor irmão chegou!<br>Use a Dança (D)!", 4000);
        }
        if (d >= 500 && !companions.find(c => c.type === 'daphy')) {
            companions.push(new Companion('daphy', 36)); lives = 5; showCustomAlert("💖 Sua pessoa especial Daphy está aqui!");
        }
        if (d >= 1200 && !companions.find(c => c.type === 'lucas')) {
            companions.push(new Companion('lucas', 48)); lives++;
            colFly.style.display = 'flex'; btnFlyMobile.style.display = 'flex';
            showCustomAlert("👼 Seu guardião Lucas chegou!<br>Aperte 'L' para Voar!", 5000);
        }
        // Transição para Surubim
        if (d >= 1600 && currentMap === 'avenida') {
            currentMap = 'surubim';
            bgTrees = []; // Remove árvores antigas
            showCustomAlert("☁️ Chegamos em Surubim!<br>Correndo pelo céu!", 6000);
        }

        if (currentEventIndex < storyEvents.length && d >= storyEvents[currentEventIndex].d) {
            triggerEvent(storyEvents[currentEventIndex]); currentEventIndex++;
        }
    }

    function updatePowers() {
        // Lógica Dança
        if (isDancing) {
            danceTimer--; fillDance.style.width = (danceTimer / POWER_TIME * 100) + '%'; fillDance.style.background = '#e74c3c'; lblDance.innerText = "Dançando! " + Math.ceil(danceTimer/60) + "s";
            if (danceTimer <= 0) { isDancing = false; danceCooldown = MAX_COOLDOWN; }
        } else if (danceCooldown > 0) {
            danceCooldown--; fillDance.style.width = ((MAX_COOLDOWN - danceCooldown) / MAX_COOLDOWN * 100) + '%'; fillDance.style.background = '#f1c40f'; lblDance.innerText = "Carregando...";
        } else if (companions.find(c => c.type === 'mango')) {
            fillDance.style.width = '100%'; fillDance.style.background = '#2ecc71'; lblDance.innerText = "Dança PRONTA (D)";
        }

        // Lógica Voo
        if (isFlying) {
            flyTimer--; fillFly.style.width = (flyTimer / POWER_TIME * 100) + '%'; fillFly.style.background = '#0984e3'; lblFly.innerText = "Voando! " + Math.ceil(flyTimer/60) + "s";
            if (flyTimer <= 0) { isFlying = false; flyCooldown = MAX_COOLDOWN; }
        } else if (flyCooldown > 0) {
            flyCooldown--; fillFly.style.width = ((MAX_COOLDOWN - flyCooldown) / MAX_COOLDOWN * 100) + '%'; fillFly.style.background = '#f1c40f'; lblFly.innerText = "Carregando...";
        } else if (companions.find(c => c.type === 'lucas')) {
            fillFly.style.width = '100%'; fillFly.style.background = '#0984e3'; lblFly.innerText = "Voo PRONTO (L)";
        }
    }

    function checkCollisions() {
        obstacles.forEach((o, i) => {
            let collision = false;
            if (player && !isFlying && !isDancing) {
                let pLeft = player.x - 15; let pRight = player.x + 15; let pTop = player.y - player.height; let pBottom = player.y;
                let oLeft = o.x - o.width/2 + 5; let oRight = o.x + o.width/2 - 5; let oTop = o.y - o.height; let oBottom = o.y;
                if (pRight > oLeft && pLeft < oRight && pBottom > oTop && pTop < oBottom) collision = true;
            }
            if (collision && !isInvincible) {
                obstacles.splice(i, 1); lives--;
                if (lives <= 0) gameOver(); else { isInvincible = true; invincibilityTimer = 90; }
            }
            if (o.x < -100) obstacles.splice(i, 1);
        });
    }

    function update() {
        frames++; distance += gameSpeed / 60; distVal.innerText = Math.floor(distance); livesVal.innerText = lives;
        if (frames % 1000 === 0) gameSpeed += 0.3; 
        if (player) player.update();

        clouds.forEach((c, i) => { c.update(); if (c.x < -150) clouds.splice(i, 1); });
        bgTrees.forEach((t, i) => { t.update(); if (t.x < -100) bgTrees.splice(i, 1); });
        
        if(currentMap === 'surubim') {
            cityBuildings.forEach(b => { b.x -= gameSpeed * 0.1; if(b.x < -100) b.x = WIDTH + 50; });
        }

        obstacles.forEach(o => o.update());
        checkCollisions();

        if (isInvincible && !isDancing && !isFlying) { invincibilityTimer--; if (invincibilityTimer <= 0) isInvincible = false; }
        updatePowers(); handleProgression();
    }

    function drawMap() {
        // Céu
        let skyGradient = ctx.createLinearGradient(0, 0, 0, HEIGHT);
        if (currentMap === 'surubim') {
            skyGradient.addColorStop(0, '#0a3d62'); skyGradient.addColorStop(1, '#82ccdd');
        } else {
            skyGradient.addColorStop(0, '#0fb9b1'); skyGradient.addColorStop(1, '#81ecec');
        }
        ctx.fillStyle = skyGradient; ctx.fillRect(0, 0, WIDTH, HEIGHT);
        
        clouds.forEach(c => c.draw());

        if (currentMap === 'surubim') {
            // Cidade ao fundo
            cityBuildings.forEach(b => { ctx.fillStyle = b.color; ctx.fillRect(b.x, HEIGHT - b.height - 20, b.width, b.height); });
            // Nuvens como pista
            ctx.fillStyle = '#ffffff'; ctx.beginPath(); ctx.ellipse(WIDTH/2, 350, WIDTH, 80, 0, 0, Math.PI*2); ctx.fill();
        } else {
            bgTrees.forEach(t => t.draw());
            ctx.fillStyle = '#0097e6'; ctx.fillRect(0, 160, WIDTH, 20); ctx.fillStyle = '#0080c0'; ctx.fillRect(0, 177, WIDTH, 3); 
            ctx.fillStyle = '#44bd32'; ctx.fillRect(0, 180, WIDTH, 20); ctx.fillStyle = '#2c851e'; ctx.fillRect(0, 197, WIDTH, 3); 
            ctx.fillStyle = '#353b48'; ctx.fillRect(0, 200, WIDTH, 150);
            ctx.fillStyle = '#f5f6fa';
            for(let i=0; i<15; i++) { ctx.fillRect((i * 80) - (frames * gameSpeed % 80), 270, 30, 4); }
            ctx.fillStyle = '#44bd32'; ctx.fillRect(0, 350, WIDTH, 20); ctx.fillStyle = '#2c851e'; ctx.fillRect(0, 350, WIDTH, 3); 
            ctx.fillStyle = '#c23616'; ctx.fillRect(0, 370, WIDTH, 30);
        }
    }

    function draw() {
        ctx.clearRect(0, 0, WIDTH, HEIGHT); drawMap();
        obstacles.forEach(o => o.draw());
        if (player) {
            [...companions].reverse().forEach(c => c.draw(player.history));
            player.draw();
        } else { drawChar('mahalia', WIDTH/2, 320, false, frames); }
    }

    function gameLoop() {
        if (gameState === 'PLAYING') { spawn(); update(); }
        if (gameState === 'PLAYING' || gameState === 'CHOICE') { draw(); requestAnimationFrame(gameLoop); }
    }

    function gameOver() {
        gameState = 'GAMEOVER'; document.getElementById('main-overlay').classList.remove('hidden');
        document.getElementById('msgTitle').innerText = "Fim da Corrida!"; document.getElementById('msgTitle').style.color = "#e74c3c";
        document.getElementById('msgText').innerText = `Vocês chegaram a ${Math.floor(distance)} metros!`;
    }

    function triggerDance() {
        if (companions.find(c => c.type === 'mango') && danceCooldown <= 0 && !isDancing && !isFlying) {
            isDancing = true; danceTimer = POWER_TIME; 
        }
    }

    function triggerFly() {
        if (companions.find(c => c.type === 'lucas') && flyCooldown <= 0 && !isFlying && !isDancing) {
            isFlying = true; flyTimer = POWER_TIME; player.jump(); // Dá um impulso ao decolar
        }
    }

    // Controles PC
    window.addEventListener('keydown', e => {
        if (gameState === 'PLAYING' && player) {
            if ((e.code === 'Space' || e.code === 'ArrowUp') && !isFlying) player.jump();
            if (e.code === 'ArrowDown') player.duck(true);
            if (e.code === 'KeyD') triggerDance();
            if (e.code === 'KeyL') triggerFly();
        } else if (e.code === 'Space' && (gameState === 'START' || gameState === 'GAMEOVER')) { startGame(); }
    });
    window.addEventListener('keyup', e => { if (player && e.code === 'ArrowDown') player.duck(false); });

    // Controles Mobile (Toque)
    const addTouch = (btn, actDown, actUp) => {
        btn.addEventListener('touchstart', e => { e.preventDefault(); if(gameState === 'PLAYING') actDown(); });
        if(actUp) btn.addEventListener('touchend', e => { e.preventDefault(); actUp(); });
    };
    addTouch(btnJump, () => { if(player && !isFlying) player.jump(); });
    addTouch(btnDuck, () => { if(player) player.duck(true); }, () => { if(player) player.duck(false); });
    addTouch(btnDance, triggerDance);
    addTouch(btnFlyMobile, triggerFly);

    document.getElementById('startBtn').addEventListener('click', startGame);

    function startGame() { init(); gameState = 'PLAYING'; document.getElementById('main-overlay').classList.add('hidden'); gameLoop(); }
    draw(); 
</script>
